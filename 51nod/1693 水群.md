#include<bits/stdc++.h>
using namespace std;
const int N = 1e6+10;
#define inf 0x7f7f7f7f

int prime[]= {2,3,5,7,11};

bool vis[N];
int dist[N];
int n,maxn;

int spfa(int s,int e)
{
    queue<int> sk;
    maxn = n+10;
    for(int i=0; i<maxn; i++)
    {
        dist[i] = inf;
        vis[i] = false;
    }
    dist[s] = 0;
    vis[s] = true;
    sk.push(s);
    while(!sk.empty())
    {
        int u = sk.front();
        sk.pop();
        vis[u] = false;
        if(u-1>0&&dist[u-1]>dist[u]+1)
        {
            dist[u-1] = dist[u] + 1;
            if(!vis[u-1])
            {
                vis[u-1] = true;
                sk.push(u-1);
            }
        }
        for(int i=0; i<5&&prime[i]*u<maxn; i++)
        {
            int to = prime[i]*u;
            int cost = prime[i];
            if(dist[to]>dist[u]+cost)
            {
                dist[to] = dist[u]+cost;
                if(!vis[to])
                {
                    vis[to]=true;
                    sk.push(to);
                }
            }
        }
    }
    return dist[e];
}


int main()
{
    scanf("%d",&n);
    printf("%d\n",spfa(1,n));
    return 0;
}
